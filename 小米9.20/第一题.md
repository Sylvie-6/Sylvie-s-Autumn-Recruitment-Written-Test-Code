```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int T;
    if (!(cin >> T)) return 0;
    while (T--) {
        int n, k;
        cin >> n >> k;
        vector<long long> a(n + 1);
        for (int i = 1; i <= n; ++i) cin >> a[i];

        // 求第 k 大阈值 L
        vector<long long> b(a.begin() + 1, a.end());
        sort(b.begin(), b.end());                      // 升序
        long long L = b[n - k];                        // 第 k 大

        // 按字典序最小的方式切分：遇到 >= L 的尽早切，切出 k-1 刀
        vector<pair<int,int>> ans;
        int l = 1, cuts = k - 1;
        for (int i = 1; i <= n; ++i) {
            if (cuts > 0 && a[i] >= L) {
                ans.push_back({l, i});
                l = i + 1;
                --cuts;
            }
        }
        ans.push_back({l, n});

        // 输出
        for (auto &p : ans) {
            cout << p.first << ' ' << p.second << "\n";
        }
    }
    return 0;
}
```
